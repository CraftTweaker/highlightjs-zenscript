    import mods.modularmachinery.RecipeBuilder;
import mods.modularmachinery.RecipePrimer;
import crafttweaker.item.IItemStack;

val furnace_recipes = furnace.all;
val furnace_values = [ [ 180, 1, 20 ], [ 160, 1, 40 ], [ 160, 2, 80 ], [ 80, 2, 100 ] ] as int[][];
val furnace_names = [ "mmd_redstone_furnace", "mmd_advanced_furnace", "mmd_arc_furnace", "mmd_superheated_furnace" ] as string[];

for rec in furnace_recipes {
    for idx, entry in furnace_values {
        val machine_n as string = furnace_names[idx];
        val input_name as string = rec.input.name;
        val output_name as string = rec.output.name;
        val rec_n as string = input_name ~ "_to_" ~ output_name ~ "_for_" ~ machine_n;
        val time as int = entry[0];
        val ioamount as int = entry[1];
        val energy as int = entry[2];
        val inputItem as IItemStack = rec.input.withAmount(ioamount);
        val outputItem as IItemStack = rec.output.withAmount(ioamount);
        val rec_p as RecipePrimer = RecipeBuilder.newBuilder(rec_n, machine_n, time);
        rec_p.addItemInput(inputItem);
        rec_p.addItemOutput(outputItem);
        rec_p.addEnergyPerTickInput(energy);
        rec_p.build();
    }
}
